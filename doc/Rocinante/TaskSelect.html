def get_task(time, priority)
  GET_TASK = "SELECT " +
            " TASK_ID," +
            " ACTIOION_ID," +
            " ACTIOION_FILE," +
            " ACTIOION_NAME," +
            " EXEC_SPAN " +
            "FROM RC_TASK_DETAIL " +
            "WHERE" +
            " (START_YEAR <= ?) AND" +
            " (START_MONTH <= ?) AND" +
            " (START_DAY <= ?) AND" +
            " (START_HOUR <= ?) AND" +
            " (START_MINUTES <= ?) AND" +
            " (TASK_STATE = 0) AND" +
            " (PRIORITY = ?) " +
            "ORDER BY UPDATE_DATE ASC "
            "LIMIT 1"

  # paramsを設定
  params = Array.new
  params.push = time.strftime(%)
  params.push = time.strftime(%)
  params.push = time.strftime(%)
  params.push = time.strftime(%)
  params.push = time.strftime(%)
  params.push = priority.to_s

  # ＳＱＬを実行
  return $memory.execute(GET_TASK, params)
end
