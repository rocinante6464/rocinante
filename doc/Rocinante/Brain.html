def do_think(think_info)
  begin
    # 思考ファイル読み込む
    load $think_root + think_info[:file_name]
    # 思考結果を返却する
    think = Kernel.const_get(think_info[:think_id]).new(@mind)

    return think
  rescue => e
    # ログの出力を行います。
    puts think_info[:think_id] + "の実行に失敗しました。エラーメッセージ：" + e.message
    @log.error(think_info[:think_id] + "の実行に失敗しました。エラーメッセージ：" + e.message)

    return false
  end
end
