def start_timer(timerSpan, observer)
  # 通知があった時に通知する先を定義
  add_observer(observer)
  # ログを設定
  @log.info("NOTIFIER:" << @notifier.to_s << " TIMER_SPAN:" << timerSpan.to_s << " >> START")
  # 通知者の実行状態
  @active = true
  # 時間の監視を開始
  time_notify(timerSpan)
end
